}

1. Добавить `"use server"` функцию `cancelOrder` в `src/actions/orders.ts`:

```typescript
export async function cancelOrder(orderId: string) {
  const session = await auth()
  if (!session?.user?.id) throw new Error("Не авторизован")

  const order = await db.customOrder.findUnique({
    where: { id: orderId },
    include: {
      items: { include: { serialUnit: true } },
      payments: true,
    },
  })
  if (!order) throw new Error("Заказ не найден")
  if (order.status === "COMPLETED") {
    throw new Error("Завершённый заказ нельзя отменить, используйте возврат")
  }
  if (order.status === "CANCELLED") {
    throw new Error("Заказ уже отменён")
  }

  await requirePermission("orders.manage", order.storeId)

  await db.$transaction(async (tx) => {
    // 1. Удалить платежи
    if (order.payments.length > 0) {
      await tx.payment.deleteMany({ where: { customOrderId: orderId } })
      await tx.customOrder.update({
        where: { id: orderId },
        data: { prepaidAmount: 0 },
      })
    }

    // 2. Вернуть серийники в IN_STOCK
    for (const item of order.items) {
      if (item.serialUnitId) {
        await tx.serialUnit.update({
          where: { id: item.serialUnitId },
          data: { status: "IN_STOCK" },
        })
      }
    }

    // 3. Удалить долг поставщику (если есть)
    const debt = await tx.supplierDebt.findFirst({ where: { orderId } })
    if (debt) {
      await tx.supplierDebt.delete({ where: { id: debt.id } })
    }

    // 4. Статус CANCELLED
    await tx.customOrder.update({
      where: { id: orderId },
      data: { status: "CANCELLED" },
    })

    // 5. Запись в историю
    await tx.orderStatusHistory.create({
      data: {
        orderId,
        status: "CANCELLED",
        comment: `Заказ отменён. Откачено: ${order.payments.length} платеж(ей), ${order.items.filter(i => i.serialUnitId).length} серийник(ов)`,
        userId: session.user!.id,
      },
    })
  })

  revalidatePath("/orders")
  return { success: true }
}
```

```typescript
/** Вычисляет ожидаемую сумму наличных в кассе на конец смены */
export async function calculateExpectedCash(
  shiftId: string,
  openingCash: number,
  tx: PrismaTx
): Promise<number> {
  const cashIncome = await tx.payment.aggregate({
    where: { shiftId, method: "CASH", isExpense: false },
    _sum: { amount: true },
  })
  const cashExpenses = await tx.payment.aggregate({
    where: { shiftId, method: "CASH", isExpense: true },
    _sum: { amount: true },
  })
  const deposits = await tx.payment.aggregate({
    where: { shiftId, method: "CASH", purpose: "DEPOSIT" },
    _sum: { amount: true },
  })
  const withdrawals = await tx.payment.aggregate({
    where: { shiftId, method: "CASH", purpose: "WITHDRAWAL" },
    _sum: { amount: true },
  })
  const cashRefunds = await tx.payment.aggregate({
    where: { shiftId, method: "CASH", purpose: "REFUND" },
    _sum: { amount: true },
  })

  return +(
    openingCash +
    Number(cashIncome._sum.amount ?? 0) +
    Number(deposits._sum.amount ?? 0) -
    Number(cashExpenses._sum.amount ?? 0) -
    Number(withdrawals._sum.amount ?? 0) -
    Number(cashRefunds._sum.amount ?? 0)
  ).toFixed(2)
}
```

   - Add function:
     ```typescript
     export function calcBankingFee(
       amount: DecimalLike,
       rate: DecimalLike,
     ): { fee: Prisma.Decimal; total: Prisma.Decimal } {
       const a = toMoney(amount)
       const r = toMoney(rate)
       if (a.isZero()) return { fee: new Prisma.Decimal(0), total: new Prisma.Decimal(0) }
       if (r.gte(1)) throw new Error("calcBankingFee: rate must be less than 1")
       if (r.isZero()) return { fee: new Prisma.Decimal(0), total: a }
       // Reverse percentage: fee = amount / (1 - rate) - amount
       const oneMinusRate = new Prisma.Decimal(1).sub(r)
       const total = a.div(oneMinusRate)
       const fee = total.sub(a)
       return {
         fee: new Prisma.Decimal(fee.toFixed(2)),
         total: new Prisma.Decimal(total.toFixed(2)),
       }
     }
     ```

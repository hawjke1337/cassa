       ```typescript
       export async function getSupplierDebtsReport(params: {
         storeId: string
         supplierId?: string
         isPaid?: boolean
         dateFrom?: string
         dateTo?: string
       }) {
         await requirePermission("orders.costs", params.storeId)
         
         const where: Prisma.SupplierDebtWhereInput = {
           order: { storeId: params.storeId },
           ...(params.supplierId ? { supplierId: params.supplierId } : {}),
           ...(params.isPaid !== undefined ? { isPaid: params.isPaid } : {}),
           ...(params.dateFrom || params.dateTo ? {
             createdAt: {
               ...(params.dateFrom ? { gte: new Date(params.dateFrom) } : {}),
               ...(params.dateTo ? { lte: new Date(params.dateTo + "T23:59:59") } : {}),
             },
           } : {}),
         }
         
         const [debts, totals, paidTotal] = await Promise.all([
           db.supplierDebt.findMany({
             where,
             include: {
               supplier: { select: { name: true } },
               order: { select: { number: true, totalAmount: true, purchasePrice: true, deliveryCost: true } },
             },
             orderBy: { createdAt: "desc" },
           }),
           db.supplierDebt.aggregate({
             where: { ...where, isPaid: false },
             _sum: { amount: true },
             _count: true,
           }),
           db.supplierDebt.aggregate({
             where: { ...where, isPaid: true },
             _sum: { amount: true },
             _count: true,
           }),
         ])
         
         return {
           debts: debts.map(d => ({
             id: d.id,
             supplierName: d.supplier.name,
             orderNumber: d.order.number,
             amount: Number(d.amount),
             isPaid: d.isPaid,
             paidAt: d.paidAt?.toISOString() ?? null,
             createdAt: d.createdAt.toISOString(),
             comment: d.comment,
           })),
           totalUnpaid: Number(totals._sum.amount ?? 0),
           unpaidCount: totals._count,
           totalPaid: Number(paidTotal._sum.amount ?? 0),
           paidCount: paidTotal._count,
         }
       }
       ```

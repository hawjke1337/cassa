1. В `src/actions/document-templates.ts` — добавить permission checks в `getDocumentData`:
   ```typescript
   export async function getDocumentData(type: DocumentType, entityId: string) {
     // PERM-04: Permission check based on document type
     switch (type) {
       case "SALE_RECEIPT":
       case "SALE_WARRANTY":
         // Sale receipt/warranty — requires pos.sell permission
         // Load sale to get storeId
         const sale = await db.sale.findUnique({ where: { id: entityId }, select: { storeId: true } })
         if (sale) await requirePermission("pos.sell", sale.storeId)
         break
       case "RECEIVE_DOC":
         // Stock receive document — requires inventory.receive permission
         const receive = await db.stockReceive.findUnique({ where: { id: entityId }, select: { storeId: true } })
         if (receive) await requirePermission("inventory.receive", receive.storeId)
         break
       case "WRITE_OFF_DOC":
         // Write-off document — requires inventory.writeoff permission
         const writeOff = await db.stockWriteOff.findUnique({ where: { id: entityId }, select: { storeId: true } })
         if (writeOff) await requirePermission("inventory.writeoff", writeOff.storeId)
         break
       case "REPAIR_RECEIPT":
       case "REPAIR_COMPLETION":
         // Repair docs — requires repairs.view permission
         const repair = await db.repair.findUnique({ where: { id: entityId }, select: { storeId: true } })
         if (repair) await requirePermission("repairs.view", repair.storeId)
         break
       case "TRADE_IN_RECEIPT":
         const tradeIn = await db.tradeIn.findUnique({ where: { id: entityId }, select: { storeId: true } })
         if (tradeIn) await requirePermission("tradein.view", tradeIn.storeId)
         break
       default:
         // Unknown type — require admin permission
         await requirePermission("settings.templates")
     }
     // ... rest of existing function logic unchanged
   }
   ```

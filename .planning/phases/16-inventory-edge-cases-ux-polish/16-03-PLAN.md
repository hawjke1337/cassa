   ```typescript
   import type { Payment, PaymentMethod } from '@prisma/client'
   import { toNumber } from '@/lib/money'
   
   export function aggregatePaymentsByMethod(
     payments: Array<{ method: PaymentMethod; amount: number | Decimal; isTradeIn?: boolean }>
   ): Array<{ method: PaymentMethod; amount: number; label: string }> {
     // Trade-in keep separate
     const tradeIn = payments.filter(p => p.isTradeIn)
     const regular = payments.filter(p => !p.isTradeIn)
     const grouped = new Map<PaymentMethod, number>()
     for (const p of regular) {
       grouped.set(p.method, (grouped.get(p.method) ?? 0) + toNumber(p.amount))
     }
     const methodOrder: PaymentMethod[] = ['CASH', 'CARD', 'SBP', 'TRANSFER', 'CREDIT']
     const result = methodOrder
       .filter(m => grouped.has(m))
       .map(m => ({ method: m, amount: grouped.get(m)!, label: METHOD_LABELS[m] }))
     if (tradeIn.length > 0) {
       result.push({
         method: 'TRADE_IN' as PaymentMethod,
         amount: tradeIn.reduce((s, p) => s + toNumber(p.amount), 0),
         label: 'Trade-in'
       })
     }
     return result
   }
   ```
